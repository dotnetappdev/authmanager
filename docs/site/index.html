// Your existing Identity setup — unchanged
builder.Services.AddDbContext<AppDbContext>(o => o.UseSqlite("..."));
builder.Services.AddIdentity<ApplicationUser, IdentityRole>()
    .AddEntityFrameworkStores<AppDbContext>();

Quick Start

AuthManager works with your existing DbContext — no separate setup needed.

1

Install

dotnet add package DotNetAuthManager
2

Keep your existing DbContext & Identity

// Unchanged — any provider (SQL Server, PostgreSQL, MySQL, SQLite…)
builder.Services.AddDbContext<AppDbContext>(o =>
    o.UseSqlite(builder.Configuration.GetConnectionString("Default")!));

builder.Services.AddIdentity<ApplicationUser, IdentityRole>()
    .AddEntityFrameworkStores<AppDbContext>()
    .AddDefaultTokenProviders();
3

Add AuthManager & seed the admin

builder.Services.AddAuthManager<ApplicationUser>(options => {
    options.RoutePrefix   = "authmanager";
    options.DefaultTheme  = AuthManagerTheme.Dark;
    options.SuperAdminRole = "SuperAdmin";
});

var app = builder.Build();

// Creates the SuperAdmin role + user on first run. Idempotent.
await app.CreateDefaultSuperUserAsync<ApplicationUser>(
    email:    "superadmin@example.com",
    password: "SuperAdmin@123456!"
);

app.UseAuthentication();
app.UseAuthorization();
app.MapAuthManager();
app.Run();
4

Open the dashboard & change your password

Navigate to https://your-app/authmanager and sign in with the seeded SuperAdmin account. Change the password immediately.

Publishing to NuGet

Pack locally, test, then push to NuGet.org or GitHub Packages.

1

Build & pack

# Build Release
dotnet build -c Release

# Pack all projects — output lands in ./nupkg/
dotnet pack src/AuthManager.Core/AuthManager.Core.csproj             -c Release -o ./nupkg
dotnet pack src/AuthManager.UI/AuthManager.UI.csproj                 -c Release -o ./nupkg
dotnet pack src/AuthManager.AspNetCore/AuthManager.AspNetCore.csproj -c Release -o ./nupkg
2

Test locally

# Register the local folder as a NuGet source
dotnet nuget add source ./nupkg --name local-authmanager

# Install in any test project
dotnet add package DotNetAuthManager --source local-authmanager

# Remove when done
dotnet nuget remove source local-authmanager
3

Push to NuGet.org

dotnet nuget push ./nupkg/DotNetAuthManager.*.nupkg \
  --api-key $NUGET_API_KEY \
  --source https://api.nuget.org/v3/index.json \
  --skip-duplicate
4

Or push to GitHub Packages

dotnet nuget add source \
  --username YOUR_GITHUB_USERNAME \
  --password $GITHUB_TOKEN \
  --store-password-in-clear-text \
  --name github \
  "https://nuget.pkg.github.com/dotnetappdev/index.json"

dotnet nuget push ./nupkg/DotNetAuthManager.*.nupkg \
  --source github --skip-duplicate
